# 环境搭建

# 初始化 package.json

npm init

# 安装 Jest

npm install jest

# 测试文件格式

xxx.test.js

# package.json 文件中配置测试执行命令

"scripts": {
    "test": "jest",
  },

# package.json 文件中配置 热更新

"scripts": {
    "test": "jest --watchAll",
  },

# package.json 文件中配置 覆盖率

"scripts": {
    "test": "jest --coverage",
  },

# 生成 jest 初始化配置

npx jest --init

注意

会生成 jest.config.js

# jest.config.js 中 配置说明

字段名 值 作用
clearMocks true/false 测试完成后清除模拟调用的东西
coverageDirectory coverage（生成代码覆盖率的文件夹名） 开启代码覆盖率
browser true/false 是否运行在浏览器环境

# 编写一个简单的测试用例

test( instructions , Arrow function )

参数名 作用
instructions 对该测试用例的描述
Arrow function 箭头函数，函数内编写测试代码
test("第一个方法", () => {
  expect(func(xxx)).toBe(xxx);
});

代码含义解析

表示对 func 方法调用传入参数 xxx 希望得到的结果为 xxx

参数名 含义
expect 期望
toBe 预期希望得到的值
func 被测试的方法名

# 匹配器

# toBe

相当于 === 匹配

test("toBe", () => {
  expect("第一个测试").toBe("第一个测试");
});

# toEqual

与 toBe 类似，唯一区别是没有 toBe 严谨，不会比较引用地址

test("toEqual", () => {
  const a = { number: "lqy" };
  expect(a).toBe({ number: "lqy" });
});

# toBeNull

比较 null 值

test("toBeNull", () => {
  const a = null;
  expect(a).toBeNull();
});

# toBeUndefined

比较 undefined 值

test("toBeUndefined", () => {
  const a = undefined;
  expect(a).toBeUndefined();
});

# toBeDefined

比较 是否被定义过

test("toBeDefined", () => {
  const a = "lqy";
  expect(a).toBeDefined();
});

# toBeTruthy

比较 是否为真值

test("toBeTruthy", () => {
  const a = 1000;
  expect(a).toBeTruthy();
});

# toBeFalsy

比较 是否为假值

test("toBeFalsy", () => {
  const a = 1000;
  expect(a).toBeFalsy();
});

# toBeGreaterThan

比较 是否 大于

test("toBeGreateThan", () => {
  // >
  const count = 10;
  expect(count).toBeGreaterThan(9);
});

# toBeLessThan

比较 是否 小于

test("toBeLessThan", () => {
  // <
  const count = 10;
  expect(count).toBeLessThan(11);
});

# toBeGreaterThanOrEqual

比较是否 大于等于

test("toBeGreaterThanOrEqual", () => {
  // >=
  const count = 10;
  expect(count).toBeGreaterThanOrEqual(10);
});

# toBeLessThanOrEqual

比较是否 小于等于

test("toBeLessThanOrEqual", () => {
  // <=
  const count = 10;
  expect(count).toBeLessThanOrEqual(11);
});

# toBeCloseTo

比较是否相等，解决 js 精度问题

test("toBeCloseTo", () => {
  const one = 0.1;
  const two = 0.2;
  expect(one + two).toBeCloseTo(0.3);
});

# toMatch

检测字符串中是否存在指定的字符串

test("toMatch", () => {
  const str = "Hello world";

  expect(str).toMatch("e");
});

# toContain

检测数组中是否存在 子项 与 指定 相等

test("toContain", () => {
  const arr = ["H", "e", "l", "l", "o"];
  expect(arr).toContain("e");
});

扩展

toContain 匹配器可以对 Set 类型数据进行检测。

# throwNewErrorFunc

异常检测匹配器

const throwNewErrorFunc = () => {
  throw new Error("this is Error");
};
test("toThrow", () => {
  expect(throwNewErrorFunc).toThrow("this is Error");
});

解析

toThrow 表示 检测测试的 模块 是否抛出了 异常信息为 Error 的异常。
toThrow() 中可以不传入参数，如果传入表示 抛出的异常信息要与参数内的内容相同。

const throwNewErrorFunc = () => {
  throw new Error("this is Error");
};
test("toThrow", () => {
  expect(throwNewErrorFunc).not.toThrow();
});

解析

not 取反操作，可以理解为 js 中的 ! 符号。

# toMatchObject

比较 目标对象中是否存在指定的数据内容

test("toMatchObject", () => {
  const obj = {
    success: true,
  };
  expect(obj).toMatchObject({
    success: true,
  });
});

# jest 异步代码测试

导读

本节为了异步操作使用了第三方库 axios，有 axios 基础的小伙伴，阅读起来会更加便于理解。

# done 方式解决 异步问题

异步方法文件

import axios from "axios";

export const fetchData = (fn) => {
  axios.get("xxx").then((response) => {
    fn(response.data);
  });
};

异步测试文件

import { fetchData } from "./fetchData.js";

test("FetchData", (done) => {
  fetchData((data) => {
    expect(data).toEqual({
      success: true,
    });
    done();
  });
});

注意

这里出现的 done 方法，表示当异步处理结束后，再调用 done() 进行 jest 测试。
不然会再异步还没结束就进行了 jest 测试。

# return 方式解决异步问题

异步方法文件

import axios from "axios";

export const fetchData = () => {
  return axios.get("xxx");
};

异步测试文件

test("FetchData", () => {
  return fetchData().then((response) => {
    expect(response.data).toEqual({
      success: true,
    });
  });
});

注意

这里出现的 return ，表示当异步处理结束后，再进行 jest 测试。
不然会再异步还没结束就进行了 jest 测试。

# 接口不存在

异步方法文件

import axios from "axios";

export const fetchData = () => {
  return axios.get("xxx");
};

异步测试文件

test("FetchData", () => {
  expect.assertions(1); // 断言，表示 必须执行一次 expect
  return fetchData().catch((error) => {
    expect(e.toString().indexOf("404") > 1).toBe(true);
  });
});

::: 解析 因为当接口存在的时候，不会触发 catch 回调
所以必须要求执行一次 expect 来进行检测 :::

# async await 方式

异步方法文件

import axios from "axios";

export const fetchData = () => {
  return axios.get("xxx");
};

异步测试文件

// 写法 一
test("FetchData", () => {
    await expect(fetchData()).resolves.toMatchObject({
        data:{
            success:true
        }
    })
});
// 写法 二
test("FetchData", () => {
    const response = await fetchData();
    expect(response.data).toEqual({
        success:true
    })
});

# jest 分组

describe("xxx", () => {});

解析

describe 的作用与 test 相同的，第一个参数是分组的描述，之后的回调函数中 编写 test 测试代码。

# jest 的四个钩子函数

# beforeAll

执行再所有测试用例之前

beforeAll(() => {
  // 代码部分
});

# afterAll

执行再所有测试用例之后

afterAll(() => {
  // 代码部分
});

# beforeEach

执行每一次测试用例之前

beforeEach(() => {
  // 代码部分
});

# afterEach

执行每一次测试用例之后

afterEach(() => {
  // 代码部分
});

# jest 钩子函数作用域

# 钩子函数作用域三大特点

  • 钩子函数在父级分组可作用域子集，类似继承
  • 钩子函数同级分组作用域互不干扰，各起作用
  • 先执行外部的钩子函数，再执行内部的钩子函数

# test 的 only 标识符

test.only("xxx", () => {
  // 代码部分
});

解析

only 表示只检测当前测试代码，其余的先跳过。
