# Observer 详细解析

# Observer 实例化

/**
 * 附加到每个观察者的观察者类
 * 对象。附加后，观察者将转换目标
 * 将对象的属性键插入获取器/设置器
 * 收集依赖关系并调度更新。
 */
var Observer = function Observer(value) {
  this.value = value; // 将要变成响应式监听的对象
  this.dep = new Dep(); // 为这个对象 创建唯一的 Dep 实例
  this.vmCount = 0;
  def(value, "__ob__", this); // 工具函数中封装的方法，用来设置定义一个属性 设置 __ob__，为不可枚举
  if (Array.isArray(value)) {
    // 判断是否是数组
    if (hasProto) {
      // 判断是否可用 __proto__
      protoAugment(value, arrayMethods); // 能使用 __proto__ 的处理
    } else {
      copyAugment(value, arrayMethods, arrayKeys); // 不能使用 __proto__ 的处理
    }
    this.observeArray(value);
  } else {
    this.walk(value);
  }
};

# 为 Observer 实例原型上绑定 walk 方法

/**
 * 浏览所有属性并将其转换为
 * getter /setter。仅在以下情况下才应调用此方法
 * 值类型为对象。
 */
Observer.prototype.walk = function walk(obj) {
  // 如果传入的对象不是数组，则获取属性集，进行循环并添加响应式
  var keys = Object.keys(obj);
  for (var i = 0; i < keys.length; i++) {

  }
};

# 为 Observer 实例原型上绑定 observeArray 方法

/**
 * 观察数组项的列表。
 */
Observer.prototype.observeArray = function observeArray(items) {
  // 创建实例时，为数组对象，则循环给每一个索引创建 Observer实例，并进行递归
  for (var i = 0, l = items.length; i < l; i++) {
    observe(items[i]);
  }
};

# 修改数组原型链，实现拦截响应式

/**
 *通过拦截来增强目标对象或数组
 *使用__proto__的原型链
 */

function protoAugment(target, src) {
  // 如果也可以使用 __proto__ 就直接将 __proto__ 替换为当前设置的原型链
  target.__proto__ = src;
}

/**
 *通过定义扩展目标对象或数组
 *隐藏的属性。
 */
function copyAugment(target, src, keys) {
  // 如果不能使用原型链
  // 则直接在数组上循环调用 def 函数 定义拦截方法属性
  // 使数组使用的时候直接使用本身的方法
  for (var i = 0, l = keys.length; i < l; i++) {
    var key = keys[i];
    def(target, key, src[key]);
  }
}

# 声明 observe 方法

/**
 * 尝试为值创建观察者实例，
 * 如果成功观察到，则返回新的观察者，
 * 或现有的观察者（如果值已包含一个）。
 */

function observe(value, asRootData) {
  // 判断 value 是对象 并且 value 不是虚拟 DOM
  if (!isObject(value) || value instanceof VNode) {
    return;
  }
  var ob;
  if (hasOwn(value, "__ob__") && value.__ob__ instanceof Observer) {
    // 判断改对象上是否存储 __ob__ 并且 __ob__ 是 Observer实例
    // 如果是 就 将 ob 直接赋值为 对象上的 __ob__
    ob = value.__ob__;
  } else if (
    // 判断 没有关闭组件观察，并且不是服务端渲染，并且是数组或者是对象 并且是否对象是可扩展可添加新属性的 并且 value不是 Vue实例
    shouldObserve &&
    !isServerRendering() &&
    (Array.isArray(value) || isPlainObject(value)) &&
    Object.isExtensible(value) &&
    !value._isVue
  ) {
    ob = new Observer(value); // 创建 Observer实例
  }
  if (asRootData && ob) {
    ob.vmCount++;
  }
  return ob; // 返回 ob 实例
}

# 声明 defineReactive$$1 方法

/**
 * 在对象上定义反应性。
 */
function defineReactive$$1(obj, key, val, customSetter, shallow) {
  // customSetter 自定义setter 方法
  // shallow true 层级浅 false 层级深
  var dep = new Dep(); // 为每一个响应式对象 添加 Dep 实例

  var property = Object.getOwnPropertyDescriptor(obj, key); // 获取指定对象上一个自有属性对应的属性描述符
  if (property && property.configurable === false) {
    // 如果存在，并且不可改写，则什么都不做
    return;
  }

  // 迎合预定义的getter /setter
  var getter = property && property.get; // 如果存在改属性，则赋值其 get 方法
  var setter = property && property.set; // 如果存在改属性，则赋值其 set 方法
  if ((!getter || setter) && arguments.length === 2) {
    // 如果 参数长度为2 并且 getter 不存在或者 setter 存在 val 赋值为 传入的 obj 的 key 属性的值
    val = obj[key];
  }

  var childOb = !shallow && observe(val); // 获取属性值的 __ob__ 属性
  Object.defineProperty(obj, key, {
    enumerable: true, // 设置可枚举
    configurable: true, // 设置可改写
    get: function reactiveGetter() {
      var value = getter ? getter.call(obj) : val; // 如果有 getter 则利用 getter 获取值，否则 使用 val
      if (Dep.target) {
        // 判断是否存在当前 watcher 实例
        dep.depend(); // 进行收集依赖
        if (childOb) {
          // 判断是否有层次
          childOb.dep.depend(); // 也为属性值添加同样的 Dep
          if (Array.isArray(value)) {
            // 判断值是否是数组
            dependArray(value);
          }
        }
      }
      return value;
    },
    set: function reactiveSetter(newVal) {
      var value = getter ? getter.call(obj) : val; // 如果有 getter 则利用 getter 获取值，否则 使用 val
      if (Dep.target) {  // 判断是否存在当前 watcher 实例
      if (newVal === value || (newVal !== newVal && value !== value)) {
        // 如果 新值等于旧值 或者 新值 和 旧值 都是 NaN 则什么都不做
        return;
      }
      if (customSetter) {
        // 如果自定义 setter 存在 则执行
        customSetter();
      }
      // 用于不带setter的访问器属性
      if (getter && !setter) {
        // 如果 getter 存在 并且 setter 不存在 则什么都不做
        return;
      }
      if (setter) {
        // 如果 setter 存在 ，则使用 setter
        setter.call(obj, newVal);
      } else {
        val = newVal;
      }
      // 获取属性值的 __ob__ 属性
      childOb = !shallow && observe(newVal);
      // 进行派发更新通知
      dep.notify();
    },
  });
}

重点备注

Vue 对数组的处理跟对象还是有挺大的不同，length 是数组的一个很重要的属性，无论数组增加元素或者删除元素（通过 splice，push 等方法操作）length 的值必定会更新，为什么不直接操作监听 length 呢？而需要拦截 splice，push 等方法进行数组的状态更新？ 原因是：在数组 length 属性上用 defineProperty 拦截的时候，会报错。

Uncaught TypeError: Cannot redefine property: length

再用 Object.getOwnPropertyDescriptor(arr, 'length')查看一下：//（Object.getOwnPropertyDescriptor 用于返回 defineProperty.descriptor） { configurable: false enumerable: false value: 0 writable: true } configurable 为 false，且 MDN 上也说重定义数组的 length 属性在不同浏览器上表现也是不一致的，所以还是老老实实拦截 splice，push 等方法，或者使用 ES6 的 Proxy。

# 声明 dependArray 方法

收集数组的依赖

/**
 * 触摸数组时收集对数组元素的依赖关系，因为
 * 我们不能像属性获取器那样拦截数组元素的访问。
 */
function dependArray(value) {
  // 循环获取数组中每一个索引
  for (var e = void 0, i = 0, l = value.length; i < l; i++) {
    e = value[i];
    // 如果这个索引里有 __ob__ 则 对其依赖收集
    e && e.__ob__ && e.__ob__.dep.depend();
    if (Array.isArray(e)) {
      dependArray(e);
    }
  }
}

# 声明 set 方法（在响应式对象上设置属性）

解析

在对象上设置一个属性。如果是新的属性就会触发更新通知，（旧属性也会触发更新通知，因为第一个添加的时候已经监听了，之后自动触发，不再手动触发）

/**
 * 在对象上设置属性。添加新属性并
 * 如果属性不触发变更通知
 * 已经存在。
 */
function set(target, key, val) {
  if (isUndef(target) || isPrimitive(target)) {
    // 判断是否为 未定义 或者 原始类型
    warn(
      "Cannot set reactive property on undefined, null, or primitive value: " +
        target
    );
    // 输出封装好的警告 不能对未定义、空值或原始值设置反应性属性:
  }
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.length = Math.max(target.length, key);
    target.splice(key, 1, val);
    return val;
  }
  if (key in target && !(key in Object.prototype)) {
    target[key] = val;
    return val;
  }
  var ob = target.__ob__;
  if (target._isVue || (ob && ob.vmCount)) {
    warn(
      "Avoid adding reactive properties to a Vue instance or its root $data " +
        "at runtime - declare it upfront in the data option."
    );
    return val;
  }
  if (!ob) {
    target[key] = val;
    return val;
  }
  defineReactive$$1(ob.value, key, val);
  ob.dep.notify();
  return val;
}

# 声明 del 方法（在响应式对象上删除属性）

解析

删除一个属性，如果必要触发通知

/**
 * 删除属性，并在必要时触发更改。
 */
function del(target, key) {
  if (isUndef(target) || isPrimitive(target)) {
    warn(
      "Cannot delete reactive property on undefined, null, or primitive value: " +
        target
    );
  }
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.splice(key, 1);
    return;
  }
  var ob = target.__ob__;
  if (target._isVue || (ob && ob.vmCount)) {
    warn(
      "Avoid deleting properties on a Vue instance or its root $data " +
        "- just set it to null."
    );
    return;
  }
  if (!hasOwn(target, key)) {
    return;
  }
  delete target[key];
  if (!ob) {
    return;
  }
  ob.dep.notify();
}
