# VNode 详细解析

# 什么是虚拟 DOM

VNode 表示 虚拟节点 Virtual DOM，为什么叫虚拟节点呢，因为不是真的 DOM 节点。

他只是用 javascript 对象来描述真实 DOM，这么描述，把 DOM 标签，属性，内容都变成 对象的属性

就像用 JavaScript 对象描述一个人一样

{sex:'男', name:'神仙朱', salary:5000,children:null}

过程就是，把你的 template 模板 描述成 VNode，然后一系列操作之后通过 VNode 形成真实 DOM 进行挂载

# 虚拟 DOM 的好处

  • 兼容性强，不受执行环境的影响。VNode 因为是 JS 对象，不管 Node 还是 浏览器，都可以统一操作， 从而获得了服务端渲染、原生渲染、手写渲染函数等能力
  • 减少操作 DOM。任何页面的变化，都只使用 VNode 进行操作对比，只需要在最后一步挂载更新 DOM，不需要频繁操作 DOM，从而提高页面性能

# Vue 中虚拟 DOM 的使用

在 Vue 中会根据模板获取真实 DOM ，再将真实 DOM 转换至抽象语法树（AST），使用抽象语法树去生成虚拟 DOM 。 在 Vue 中当数据变更后，会生成一个新的虚拟 DOM ，会经过 diff 算法，将 旧的虚拟 DOM 中变更的地方用 新的虚拟 DOM 进行替换，最后在更新页面。

# 虚拟 DOM 实例

var VNode = function VNode(
  tag, // 标签名
  data, // 存储节点的属性，class style 及属性值
  children, // 子元素（数组）
  text, // 文本内容
  elm, // 真实 Dom 节点
  context, // 上下文对象 Vue实例
  componentOptions,
  asyncFactory
) {
  this.tag = tag; // 标签名
  this.data = data;
  this.children = children; // 子元素
  this.text = text; // 文本内容
  this.elm = elm; // 真实 Dom节点
  this.ns = undefined;
  this.context = context; // 渲染这个模板的上下文对象 context 就是 Vue 实例（页面或组件）
  this.fnContext = undefined;
  this.fnOptions = undefined;
  this.fnScopeId = undefined;
  this.key = data && data.key;
  this.componentOptions = componentOptions; // 就是父组件传给子组件的自定义事件或参数，其中children 存储的为slot
  this.componentInstance = undefined; // 就是组件生成的实例
  this.parent = undefined;
  this.raw = false;
  this.isStatic = false; // 是否静态节点，如果被标记为静态节点，数据变化时将不会进行更新
  this.isRootInsert = true;
  this.isComment = false; // 是否是注释节点
  this.isCloned = false; // 是否克隆节点
  this.isOnce = false;
  this.asyncFactory = asyncFactory;
  this.asyncMeta = undefined;
  this.isAsyncPlaceholder = false;
};

# 为组件设置可改写子元素操作

var prototypeAccessors = { child: { configurable: true } };

// 向后兼容的componentInstance的别名。
prototypeAccessors.child.get = function() {
  return this.componentInstance;
};
//Object.defineProperties() 方法直接在一个对象上定义新的属性或修改现有属性，并返回该对象。
Object.defineProperties(VNode.prototype, prototypeAccessors);

# 声明 创建 虚拟 DOM 空节点

var createEmptyVNode = function(text) {
  if (text === void 0) text = "";

  var node = new VNode();
  node.text = text;
  node.isComment = true;
  return node;
};

# 声明 创建 虚拟 DOM 文本节点

function createTextVNode(val) {
  return new VNode(undefined, undefined, undefined, String(val));
}

# 声明克隆 虚拟 DOM 方法

//优化的浅表克隆
//用于静态节点和插槽节点，因为它们可以在多个节点之间重用
//多个渲染，克隆它们可以避免在DOM操作依赖时出错
//在Elm参考上。
function cloneVNode(vnode) {
  var cloned = new VNode(
    vnode.tag,
    vnode.data,
    //克隆子数组以避免在克隆的情况下变异原始数组
    //一个孩子。
    vnode.children && vnode.children.slice(),
    vnode.text,
    vnode.elm,
    vnode.context,
    vnode.componentOptions,
    vnode.asyncFactory
  );
  cloned.ns = vnode.ns;
  cloned.isStatic = vnode.isStatic;
  cloned.key = vnode.key;
  cloned.isComment = vnode.isComment;
  cloned.fnContext = vnode.fnContext;
  cloned.fnOptions = vnode.fnOptions;
  cloned.fnScopeId = vnode.fnScopeId;
  cloned.asyncMeta = vnode.asyncMeta;
  cloned.isCloned = true;
  return cloned;
}
